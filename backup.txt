function DraggableTask({ task }: any) {
  const { attributes, listeners, setNodeRef, transform } = useDraggable({
    id: task.id,
  });

  const style = {
    transform: CSS.Translate.toString(transform),
  };

  return (
    <Card
      ref={setNodeRef}
      {...listeners}
      {...attributes}
      style={style}
      sx={{
        mb: 2,
        borderRadius: 3,
        cursor: "grab",
        transition: "0.2s",
        "&:hover": {
          transform: "scale(1.02)",
          boxShadow: 6,
        },
      }}
    >
      <CardContent>
        <Typography fontWeight={700}>{task.title}</Typography>

        <Divider sx={{ my: 1 }} />

        <Typography variant="body2">{task.description}</Typography>
      </CardContent>
    </Card>
  );
}

function DroppableColumn({ status, title, tasks }: any) {
  const { setNodeRef } = useDroppable({
    id: status,
  });

  return (
    <Paper
      ref={setNodeRef}
      sx={{
        flex: 1,
        background: "#ebecf0",
        p: 2,
        borderRadius: 2,
        minHeight: 500,
      }}
    >
      <Typography variant="h6" sx={{ mb: 2 }}>
        {title}
      </Typography>

      {tasks.map((task: any) => (
        <DraggableTask key={task.id} task={task} />
      ))}
    </Paper>
  );
}

  const dispatch = useAppDispatch();
  const { tasks, loading } = useAppSelector((state) => state.userTasks);
  const router = useRouter();
  const [notifications, setNotifications] = useState<any[]>([]);
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null);

  // useEffect(() => {
  //   dispatch(fetchUserTasks());

  //   socket.connect();

  //   socket.on("taskCreated", () => {

  // }, [dispatch]);

  useEffect(() => {
    dispatch(fetchUserTasks());

    const socket = getSocket();
    socket.connect();

    socket.on("taskCreated", () => {
      dispatch(fetchUserTasks());
    });

    socket.on("taskUpdated", () => {
      dispatch(fetchUserTasks());
    });

    socket.on("notification", (data) => {
      setNotifications((prev) => [data, ...prev]);
    });

    return () => {
      socket.off("taskCreated");
      socket.off("taskUpdated");
      socket.off("notification");
    };
  }, [dispatch]);

  const handleDragEnd = (event: DragEndEvent) => {
    const { active, over } = event;

    if (!over) return;

    const taskId = Number(active.id);
    const newStatus = over.id as string;

    const task = tasks.find((t) => t.id === taskId);

    if (task && task.status !== newStatus) {
      dispatch(
        updateUserTaskStatus({
          taskId,
          status: newStatus,
        }),
      );
    }
  };

  const handleLogout = () => {
    dispatch(logout());
    router.push("/login");
  };

  return (

        position="static"
        sx={{

          width: "99%",
        }}
      >
        <Toolbar>
          <Typography sx={{ flexGrow: 1 }} variant="h6">
            User Dashboard
          </Typography>

          <IconButton
            color="inherit"
            onClick={(e) => setAnchorEl(e.currentTarget)}
          >
            <Badge badgeContent={notifications.length} color="error">
              <NotificationsIcon />
            </Badge>
          </IconButton>
          <Button
            variant="contained"
            color="error"
            onClick={handleLogout}
            sx={{ ml: "107rem", borderRadius: "3rem" }}
          >
            Logout
          </Button>
        </Toolbar>
      </AppBar>
      <Menu
        anchorEl={anchorEl}
        open={Boolean(anchorEl)}
        onClose={() => setAnchorEl(null)}
      >
        {notifications.length === 0 && <MenuItem>No Notifications</MenuItem>}

        {notifications.map((n) => (
          <MenuItem
            key={n.id}
            onClick={() => {
              setNotifications((prev) => prev.filter((x) => x.id !== n.id));
              setAnchorEl(null);
            }}
          >
            {n.message}
          </MenuItem>
        ))}
      </Menu>

      <Container maxWidth={false} sx={{ mt: 4, px: 4 }}>
        {loading && (
          <Box sx={{ textAlign: "center", mt: 5 }}>
            <CircularProgress />
          </Box>
        )}

        <DndContext
          collisionDetection={closestCenter}
          onDragEnd={handleDragEnd}
        >
          <Box
            sx={{
              display: "flex",
              gap: 2,
              width: "100%",
            }}
          >
            {Object.entries(columns).map(([status, title]) => (
              <DroppableColumn
                key={status}
                status={status}
                title={title}
                tasks={tasks.filter((task) => task.status === status)}
              />
            ))}
          </Box>
        </DndContext>
      </Container>
    </Box>
  );

import {
  Injectable,
  NotFoundException,
  ForbiddenException,
} from '@nestjs/common';

import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';

import { Task } from 'src/domain/entities/task.entity';
import { User, UserRole } from 'src/domain/entities/user.entity';

import { CreateTaskDto } from './dto/create-task.dto';
import { UpdateTaskStatusDto, TaskStatus } from './dto/update-task-status.dto';

import { ActivityLogsService } from '../activity-log/activity-log.service';
import { NotificationsService } from '../notification/notification.service';
import { EmailService } from '../email/email.service';
import { TasksGateway } from './task.gateway';

@Injectable()
export class TasksService {
  constructor(
    @InjectRepository(Task)
    private readonly taskRepo: Repository<Task>,

    @InjectRepository(User)
    private readonly userRepo: Repository<User>,

    private readonly activityLogsService: ActivityLogsService,
    private readonly notificationsService: NotificationsService,
    private readonly emailService: EmailService,
    private readonly tasksGateway: TasksGateway,
  ) {}

  async createTask(dto: CreateTaskDto, creatorId: number) {
    const creator = await this.userRepo.findOne({
      where: { id: creatorId },
    });

    if (!creator) throw new NotFoundException('Admin not found');

    if (creator.role !== UserRole.CREATOR) {
      throw new ForbiddenException('Only creators can create tasks');
    }

    const assignedUser = await this.userRepo.findOne({
      where: { id: dto.assignedToId },
    });

    if (!assignedUser) throw new NotFoundException('User not found');

    if (assignedUser.role === UserRole.CREATOR) {
      throw new ForbiddenException('Cannot assign tasks to creator');
    }

    const task = this.taskRepo.create({
      title: dto.title,
      description: dto.description,
      assignedTo: assignedUser,
      createdBy: creator,
      status: TaskStatus.PENDING,
    });

    const savedTask = await this.taskRepo.save(task);

    await this.activityLogsService.createActivityLog(
      savedTask.id,
      creator.id,
      'task_created',
      TaskStatus.PENDING,
      TaskStatus.PENDING,
    );

    await this.notificationsService.createNotification({
      userId: assignedUser.id,
      taskId: savedTask.id,
      message: `You have been assigned a new task: "${savedTask.title}"`,
    });

    try {
      await this.emailService.sendTaskAssignedEmail(
        assignedUser.email,
        assignedUser.name,
        savedTask.title,
      );
    } catch (error) {
      console.error('Email failed:', error.message);
    }

    this.tasksGateway.taskCreated(savedTask);

    return savedTask;
  }

  async getAllTasks() {
    return this.taskRepo.find({
      relations: ['assignedTo', 'createdBy'],
      order: { createdAt: 'DESC' },
    });
  }
  async updateTaskStatus(
    taskId: number,
    dto: UpdateTaskStatusDto,
    userId: number,
  ) {
    const task = await this.taskRepo.findOne({
      where: { id: taskId },
      relations: ['assignedTo', 'createdBy'],
    });

    if (!task) throw new NotFoundException('Task not found');

    if (task.assignedTo.id !== userId) {
      throw new ForbiddenException('You can only move your assigned tasks');
    }

    const oldStatus: TaskStatus = task.status;

    if (oldStatus === dto.status) {
      return task;
    }

    task.status = dto.status;

    const updatedTask = await this.taskRepo.save(task);

    await this.activityLogsService.createActivityLog(
      task.id,
      userId,
      'status_changed',
      oldStatus,
      dto.status,
    );

    const creator = await this.userRepo.findOne({
      where: { role: UserRole.CREATOR },
    });

    if (creator) {
      if (dto.status === TaskStatus.IN_REVIEW) {
        try {
          await this.emailService.sendTaskQCEmail(
            creator.email,
            task.assignedTo.name,
            task.title,
          );
        } catch (error) {
          console.error('Email QC failed:', error.message);
        }

        await this.notificationsService.createNotification({
          userId: creator.id,
          taskId: task.id,
          message: `${task.assignedTo.name} moved "${task.title}" to QC`,
        });
      }

      if (dto.status === TaskStatus.COMPLETED) {
        try {
          await this.emailService.sendTaskCompletedEmail(
            creator.email,
            task.assignedTo.name,
            task.title,
          );
        } catch (error) {
          console.error('Email completion failed:', error.message);
        }

        await this.notificationsService.createNotification({
          userId: creator.id,
          taskId: task.id,
          message: `${task.assignedTo.name} completed "${task.title}"`,
        });
      }
    }

    this.tasksGateway.taskUpdated(updatedTask);

    return updatedTask;
  }

  async getUserTasks(userId: number) {
    return this.taskRepo.find({
      where: {
        assignedTo: { id: userId },
      },
      relations: ['assignedTo', 'createdBy'],
      order: { createdAt: 'DESC' },
    });
  }
}
